package util;

public class Email {
//Email: ngocan2003krp@gmail.com |Ví dụ
//Password: iqwjfknasldniaskd	 |

public static void main(String[] args){
		
		final String from="ngocan2003krp@gmail.com";
		final String password="iqwjfknasldniaskd";
				
		//Khai báo các thuộc tính: Properties
		Properties props = new Properties();
		props.put("mail.smtp.host", "smtp.gmail.com"); //smtp host (tự tìm hiểu)
		props.put("mail.smtp.post", "587");//Thêm port: TLS 587/ SSL 456
		props.put("mail.smtp.auth", "true");//True có nghĩa khi sử dụng host được set bên trên vẫn phải đăng nhập gmail
		props.put("mail.smtp.starttls.enable", "true");

		//Create Authenticator
		Authenticator auth = new Authenticator(){
			//Vào source chọn overide, chọn getPasswordAuthentication(), sau đó overide lại return
			return new PasswordAuthentication(from,password);
		}
		
		//Tạo phiên làm việc
		Session session = Session.getInstance()

		//Gửi email
		final String to ="levumaithuy@gmail.com";

		MimeMessage msg = new MimeMessage(session);
		try{
			//Kiểu nội dung
			msg.addHeader("Content-type", "text/HTML; charset=UTF-8");
			//Người gửi
			msg.setFrom(from);
			//Ở đây sử dụng kiểu "TO" ngoài ra còn BCC, CC,... trước khi dùng lại thằng này hãy lên mạng tìm hiểu
			//Người nhận
			msg.setRecipient(Message.RecipientType.TO, InternetAddress.parse(to,false));
			//Tiêu đề email
			msg.setSubject("Email thử nghiệm");
			//Quy định ngày gửi
			msg.setSentDate(new Date());
			//Quy định email sẽ nhận reply của khách hàng
			//msg.setReplyTo(InternetAddress.parse(<Email>,false));
			//Nội dung
			msg.setText("Đây là phần nội dung", "UTF-8");//Tin nhắn bth
			msg.setContent("<Cho đoạn code html, có thể chưa link>, "text/html; charset=UTF-8");//Gửi bằng dạng code
			//Gửi email
			Transport.send(msg);
		} catch (Exception e){
			e.prin....
		}
	}
}
